pipeline 
{
    agent { label 'WinNode' }

    stages {

        stage('Clean Workspace') {
            steps {
                echo "Cleaning workspace"
                cleanWs()
            }
        }

        stage('Checkout Code') {
            steps {
                git branch: 'main',
                    url: 'https://github.com/testerintelligent/APIWEB_BDDCUCU_SELENIUM.git'
            }
        }

        stage('Build and Test') {
            steps {
                
                bat 'mvn -Dtest=TestRunner test'
            }
        }
    }

    post {

        always {
                    echo 'Publishing test and cucumber reports...'  

			sleep(time:3,unit:"SECONDS")
			cucumber buildStatus: 'UNCHANGED', 
            customCssFiles: '', 
            customJsFiles: '', 
             jsonReportDirectory: 'target/cucumber-report',
                        fileIncludePattern: 'cucumber.json',
                        reportTitle: 'Report',
            sortingMethod: 'ALPHABETICAL', 
            trendsLimit: 5

            junit '**/*.xml'

        }

       

    }
}
