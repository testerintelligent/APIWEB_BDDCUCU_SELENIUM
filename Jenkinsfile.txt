pipeline 
{
    agent { label 'WinNode' }

    stages {

        stage('Clean Workspace') {
            steps {
                echo "Cleaning workspace"
                cleanWs()
            }
        }

        stage('Checkout Code') {
            steps {
                git branch: 'main',
                    url: 'https://github.com/testerintelligent/APIWEB_BDDCUCU_SELENIUM.git'
            }
        }

        stage('Build and Test') {
            steps {
                bat 'mvn clean verify -Dtest=com.runner.RunnerClass'
            }
        }

        stage('Verify Report File') {
            steps {
                bat 'dir target\\cucumber-report'
            }
        }

    }
 
    
    post {
        
        always {
            echo 'Publishing test and cucumber reports...'


        script {
              println "Sleeping using Thread.sleep"
               Thread.sleep(8000)  // 5000 milliseconds = 5 seconds
           }
        
            
           publishHTML([allowMissing: false, alwaysLinkToLastBuild: false, icon: '', keepAll: false, reportDir: 'target/site/jacoco', reportFiles: 'index.html', reportName: 'HTML Report', reportTitles: '', useWrapperFileDirectly: true])
           
           script {
               println "Sleeping using Thread.sleep"
               Thread.sleep(8000)  // 5000 milliseconds = 5 seconds
           }


          junit '**/target/surefire-reports/*.xml'
        
            cucumber buildStatus: 'UNCHANGED',
                     jsonReportDirectory: 'target/cucumber-report',
                     fileIncludePattern: 'cucumber.json',
                     reportTitle: 'BDD Report',
                     sortingMethod: 'ALPHABETICAL',
                     trendsLimit: 5

        }

    }
}
