pipeline 
{
    agent { label 'WinNode' }

    stages {

        stage('Clean Workspace') {
            steps {
                echo "Cleaning workspace"
                cleanWs()
            }
        }

        stage('Checkout Code') {
            steps {
                git branch: 'main',
                    url: 'https://github.com/testerintelligent/APIWEB_BDDCUCU_SELENIUM.git'
            }
        }

        stage('Build and Test') {
            steps {
                
                bat 'mvn clean test -Dtest=com.runner.RunnerClass'
            }
        }


    }

    post {

        always {
                    echo 'Publishing test and cucumber reports...'  

                    script {
                        println "Sleeping using Thread.sleep"
                        Thread.sleep(15000)  // 8000 milliseconds = 15 seconds
                    }
                        cucumber buildStatus: 'null', 
                        customCssFiles: '', 
                        customJsFiles: '', 
                        failedFeaturesNumber: -1, 
                        failedScenariosNumber: -1, 
                        failedStepsNumber: -1, 
                        fileIncludePattern: '**/*.json', 
                        pendingStepsNumber: -1, 
                        skippedStepsNumber: -1, 
                        sortingMethod: 'ALPHABETICAL', 
                        undefinedStepsNumber: -1
                   
        


                    junit testResults: '**target/surefire-reports/testng-results.xml' 

        }

    }
}
